{# Troubleshooting Section #}
Common Errors & Fixes
Error: "TypeError: range undefined"

This happens if you click "Run" in the editor. Ignore it. The script only runs when you edit the Sheet.

Error: "Script function not found: doGet"

Update your code to the latest version below and Redeploy as a "New Version".

/**
 * PadosiAgent Invoice Automator - Production Script v2.2
 */
const MAIN_FOLDER_ID = "PASTE_YOUR_DRIVE_FOLDER_ID_HERE";

function doGet(e) {
  const sheetUrl = SpreadsheetApp.getActiveSpreadsheet().getUrl();
  return HtmlService.createHtmlOutput("<script>window.location.href='" + sheetUrl + "';</script>");
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    sheet.appendRow([
      data.invoice_number, 
      data.date, 
      data.agent_name + " (UID:" + data.agent_id + ")", 
      data.agent_email, 
      data.agent_mobile, 
      data.plan_name, 
      data.base_amount, 
      data.gst_amount, 
      data.total_amount, 
      data.payment_id, 
      data.promo_code || "None", 
      data.discount_folder, // Column L
      data.pdf_url           // Column M
    ]);
    return ContentService.createTextOutput(JSON.stringify({ "status": "success" })).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": err.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}

function onEdit(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const range = e.range;
  if (range.getColumn() == 12) { // Monitor 'Discount Folder' Column
    const folderName = range.getValue();
    const pdfUrl = sheet.getRange(range.getRow(), 13).getValue(); // Get PDF Link from Column M
    if (folderName && pdfUrl) processFile(folderName, pdfUrl);
  }
}

function processFile(category, fileUrl) {
  const parentFolder = DriveApp.getFolderById(MAIN_FOLDER_ID);
  let targetFolder;
  const it = parentFolder.getFoldersByName(category);
  targetFolder = it.hasNext() ? it.next() : parentFolder.createFolder(category);
  let fileId = fileUrl.includes("id=") ? fileUrl.split("id=")[1].split("&")[0] : fileUrl.split("/d/")[1].split("/")[0];
  DriveApp.getFileById(fileId).moveTo(targetFolder);
}
